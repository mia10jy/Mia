// Animate stat numbers

function animateNumber(el, target, suffix = '') {

  let start = 0;

  const isDecimal = target % 1 !== 0;

  const duration = 1500;

  const startTime = performance.now();

  function update(now) {

    const elapsed = now - startTime;

    const progress = Math.min(elapsed / duration, 1);

    const eased = 1 - Math.pow(1 - progress, 3);

    const current = isDecimal ? (start + (target - start) * eased).toFixed(1) : Math.round(start + (target - start) * eased);

    el.textContent = current + suffix;

    if (progress < 1) requestAnimationFrame(update);

  }

  requestAnimationFrame(update);

}


const statObserver = new IntersectionObserver((entries) => {

  entries.forEach(entry => {

    if (entry.isIntersecting) {

      const nums = entry.target.querySelectorAll('.stat-num');

      nums.forEach(num => {

        const text = num.textContent.trim();

        const match = text.match(/^([\d.]+)/);

        if (match) {

          const val = parseFloat(match[1]);

          const unit = num.querySelector('.stat-unit');

          const unitText = unit ? unit.outerHTML : '';

          num.innerHTML = '0' + unitText;

          setTimeout(() => {

            num.innerHTML = '0' + unitText;

            animateNumber({ 

              set textContent(v) { num.innerHTML = v + unitText; } 

            }, val);

          }, 200);

        }

      });

      statObserver.unobserve(entry.target);

    }

  });

}, { threshold: 0.5 });
